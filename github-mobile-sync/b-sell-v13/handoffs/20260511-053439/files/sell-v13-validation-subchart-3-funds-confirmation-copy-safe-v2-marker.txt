{BB趋势股卖点公式 v13 验证副图3：资金失衡与确认层}
{定位：趋势股持仓后的分层风险提示系统}
零轴:0,COLORWHITE;
版本线:0.2,COLORCYAN;
{优先级：确认层 > 结构转弱层 > 资金失衡层 > 动能衰减层 > 形态预警层}
{说明：由 v13 主公式生成，输出固定0-6资金/确认诊断层，不改变主公式}

{=== 基础参数与布林带 ===}
N:=20;
P:=2;
MID:=MA(CLOSE,N);
TMP:=STD(CLOSE,N);
UPPER:=MID+P*TMP;
LOWER:=MID-P*TMP;

{=== 均线 ===}
MA5:=MA(CLOSE,5);
MA20:=MA(CLOSE,20);
MA60:=MA(CLOSE,60);
MA120:=MA(CLOSE,120);

{=== 成交额统一口径 ===}
AMT0:=IF(AMOUNT>0,AMOUNT,VOL*CLOSE*100);
均额20:=MA(AMT0,20);

{=== 趋势与高位基础 ===}
MA60多头:=MA60>REF(MA60,5) AND MA60>MA120;
高位:=(CLOSE>MA60*1.15 OR HIGH>=HHV(HIGH,60)*0.95) AND MA60多头;
阶段涨幅充分:=CLOSE>=LLV(LOW,60)*1.45;
中期乖离充分:=CLOSE>=MA60*1.35;
背离高位基础:=阶段涨幅充分 OR 中期乖离充分;

{=== K线基础变量 ===}
实体:=ABS(CLOSE-OPEN);
上影:=HIGH-IF(CLOSE>OPEN,CLOSE,OPEN);
全幅:=HIGH-LOW+0.001;
涨跌幅:=(CLOSE-REF(CLOSE,1))/REF(CLOSE,1)*100;
收盘位置:=(CLOSE-LOW)/全幅;
上影占比:=上影/全幅;

{=== 均线排列与粘合 ===}
多头排列:=MA5>MA20 AND MA20>MA60 AND MA60>MA120;
均线最高:=IF(MA5>MA20,MA5,MA20);
均线最高2:=IF(均线最高>MA60,均线最高,MA60);
均线最高3:=IF(均线最高2>MA120,均线最高2,MA120);
均线最低:=IF(MA5<MA20,MA5,MA20);
均线最低2:=IF(均线最低<MA60,均线最低,MA60);
均线最低3:=IF(均线最低2<MA120,均线最低2,MA120);
均线粘合:=(均线最高3-均线最低3)/均线最低3<0.05;
曾经多头:=COUNT(MA5>MA20*1.02 AND MA20>MA60*1.02,30)>=10;

{=== 动能衰减层 1：量价背离 ===}
旧20日高点:=REF(HHV(HIGH,20),1);
价格新高:=HIGH>=旧20日高点;
前高距:=HHVBARS(REF(HIGH,1),20)+1;
前高成交额:=REF(AMT0,前高距);
较前高缩额:=AMT0<前高成交额*0.85;
相对均额缩:=AMT0<均额20*0.8;
前三日均额:=REF(MA(AMT0,3),1);
较近3日缩额:=AMT0<前三日均额*0.85;
基础缩额:=相对均额缩 OR 较近3日缩额;
高点缩额:=较前高缩额;
成交额萎缩:=基础缩额 AND 高点缩额;
低于均额60:=AMT0<均额20*0.6;
成交额枯竭:=低于均额60 AND 较前高缩额;
近5日额:=SUM(AMT0,5);
前5日额:=REF(SUM(AMT0,5),5);
持续缩额:=近5日额<前5日额;
背离基础:=价格新高 AND 背离高位基础 AND MA60多头;
背离强3级:=背离基础 AND 成交额枯竭 AND 持续缩额;
背离3级:=背离基础 AND 成交额萎缩 AND 持续缩额 AND 成交额枯竭=0;
背离2级:=背离基础 AND 成交额枯竭 AND 持续缩额=0;
背离1级:=背离基础 AND 成交额萎缩 AND 成交额枯竭=0 AND 持续缩额=0;
背离等级:=IF(背离强3级,4,IF(背离3级,3,IF(背离2级,2,IF(背离1级,1,0))));
近3日背离最高:=HHV(REF(背离等级,1),3);
背离显示:=背离等级>0 AND 背离等级>近3日背离最高;

{=== 动能衰减层 2：严格MACD顶背离 ===}
DIF:=EMA(CLOSE,12)-EMA(CLOSE,26);
当前价峰:=HIGH=HHV(HIGH,5);
前峰距:=HHVBARS(REF(HIGH,5),30)+5;
前峰价:=REF(HIGH,前峰距);
前峰DIF:=REF(DIF,前峰距);
价格更高:=HIGH>前峰价*1.01;
DIF均强:=MA(ABS(DIF),30);
前峰DIF有效:=前峰DIF>0 AND 前峰DIF>DIF均强*0.3;
DIF未新高:=DIF<前峰DIF*0.95;
严格MACD顶背离:=当前价峰 AND 价格更高 AND 前峰DIF有效 AND DIF未新高 AND DIF>0 AND MA60多头 AND 背离高位基础;
近5日MACD背离次数:=COUNT(REF(严格MACD顶背离,1),5);
MACD顶背离显示:=严格MACD顶背离 AND 近5日MACD背离次数=0;

{=== 资金失衡层基础 ===}
明显放额:=AMT0>均额20*1.5;

{=== 资金失衡 1：放量滞涨 ===}
近5日前收高:=REF(HHV(CLOSE,5),1);
滞涨未推进:=CLOSE<=近5日前收高*1.01;
涨势停滞:=涨跌幅>-1 AND 涨跌幅<1;
冲高回落:=上影>实体 AND 上影占比>=0.45 AND 收盘位置<=0.45;
非大涨:=涨跌幅<3;
放量滞涨:=明显放额 AND 非大涨 AND ((涨势停滞 AND 滞涨未推进) OR 冲高回落) AND 高位;
放量滞涨显示:=放量滞涨 AND COUNT(REF(放量滞涨,1),3)=0;

{=== 资金失衡 2：放量下跌 ===}
放量下跌基础:=明显放额 AND 高位 AND 涨跌幅<-2;
收盘偏低:=收盘位置<=0.35;
放量下跌:=放量下跌基础 AND 收盘偏低;
放量下跌显示:=放量下跌 AND COUNT(REF(放量下跌,1),3)=0;

{=== 资金失衡 3：强滞 ===}
三日均额:=MA(AMT0,3);
三日持续放额:=三日均额>均额20*1.35;
前高60:=REF(HHV(HIGH,60),3);
近3日高点:=HHV(HIGH,3);
靠近前高:=近3日高点>=前高60*0.97;
未有效突破前高:=近3日高点<前高60*1.02;
三日显著上推:=CLOSE>=REF(CLOSE,3)*1.05;
近3日收盘上移:=CLOSE>REF(CLOSE,1) AND REF(CLOSE,1)>REF(CLOSE,2);
近3日低点不降:=LOW>=REF(LOW,1)*0.995 AND REF(LOW,1)>=REF(LOW,2)*0.995;
三日稳步推进:=近3日收盘上移 AND 近3日低点不降;
前高附近:=COUNT(CLOSE>=前高60*0.99,3)>=2 AND CLOSE>=前高60*0.995;
前高附近推进:=前高附近 AND 近3日收盘上移 AND 近3日低点不降;
三日推进保护:=三日显著上推 OR 三日稳步推进 OR 前高附近推进;
三日推进乏力:=三日推进保护=0;
强滞:=三日持续放额 AND 靠近前高 AND 未有效突破前高 AND 三日推进乏力 AND 高位;
强滞显示:=强滞 AND COUNT(REF(强滞,1),3)=0;

{强滞代表K：最近3日最大成交额K}
强滞额最大今:=AMT0>=REF(AMT0,1) AND AMT0>=REF(AMT0,2);
强滞额最大昨:=REF(AMT0,1)>AMT0 AND REF(AMT0,1)>=REF(AMT0,2);
强滞代表高:=IF(强滞额最大今,HIGH,IF(强滞额最大昨,REF(HIGH,1),REF(HIGH,2)));
强滞代表低:=IF(强滞额最大今,LOW,IF(强滞额最大昨,REF(LOW,1),REF(LOW,2)));
强滞代表中位:=(强滞代表高+强滞代表低)/2;

{=== 资金失衡 4：天量天价 ===}
旧60日高点:=REF(HHV(HIGH,60),1);
创60日新高:=HIGH>旧60日高点;
六十日天量:=AMT0=HHV(AMT0,60);
天量天价:=六十日天量 AND 创60日新高 AND MA60多头;
天量天价显示:=天量天价 AND COUNT(REF(天量天价,1),2)=0;

{=== 资金失衡层内部互斥 ===}
强滞最终:=强滞显示;
放量下跌最终:=放量下跌显示 AND 强滞最终=0;
天量天价最终:=天量天价显示 AND 强滞最终=0 AND 放量下跌最终=0;
放量滞涨最终:=放量滞涨显示 AND 强滞最终=0 AND 放量下跌最终=0 AND 天量天价最终=0;

{=== 确认层 1：强败 ===}
强滞失败基准:=强滞最终;
距强滞:=BARSLAST(强滞失败基准);
强败窗口:=距强滞>=1 AND 距强滞<=3;
强败基准中位:=IF(距强滞=1,REF(强滞代表中位,1),IF(距强滞=2,REF(强滞代表中位,2),IF(距强滞=3,REF(强滞代表中位,3),0)));
强败基准有效:=强败窗口 AND 强败基准中位>0;
强败跌破:=强败基准有效 AND CLOSE<强败基准中位;
强败显示:=强败跌破 AND COUNT(REF(强败跌破,1),3)=0;

{=== 确认层 2：天败 ===}
天败失败基准:=天量天价最终;
距天:=BARSLAST(天败失败基准);
天败窗口:=距天>=1 AND 距天<=3;
天败中位0:=(HIGH+LOW)/2;
天败基准中位:=IF(距天=1,REF(天败中位0,1),IF(距天=2,REF(天败中位0,2),IF(距天=3,REF(天败中位0,3),0)));
天败基准有效:=天败窗口 AND 天败基准中位>0;
天败跌破:=天败基准有效 AND CLOSE<天败基准中位;
天败显示:=天败跌破 AND COUNT(REF(天败跌破,1),3)=0;

{=== 确认层内部互斥 ===}
强败最终:=强败显示;
天败最终:=天败显示 AND 强败最终=0;
确认层可显:=强败最终 OR 天败最终;

{=== 结构转弱 1：MA20结构转弱（弱） ===}
失守MA20:=CLOSE<MA20*0.99;
近3日跌破MA20:=COUNT(CROSS(MA20,CLOSE),3)>0;
跌破MA20确认:=失守MA20 AND 近3日跌破MA20;
MA5优势消失:=MA5<=MA20*1.01;
MA5下压:=MA5<REF(MA5,1);
排列走坏:=MA5优势消失 AND MA5下压;
MA20浅破:=失守MA20 AND CLOSE>=MA20*0.98;
MA20中破:=CLOSE<MA20*0.98 AND CLOSE>=MA20*0.95;
MA20深破:=CLOSE<MA20*0.95;
MA20三日非强上行:=MA20<=REF(MA20,3)*1.003;
MA5已压下:=MA5<MA20;
资金层已成:=强滞最终 OR 放量下跌最终 OR 天量天价最终 OR 放量滞涨最终;
近5日资金风险:=COUNT(REF(资金层已成,1),5)>0;
MA20破位质量:=MA20深破 OR (MA20中破 AND (MA5已压下 OR MA20三日非强上行 OR 近5日资金风险)) OR (MA20浅破 AND MA20三日非强上行);
MA20结构转弱:=跌破MA20确认 AND MA20破位质量 AND 排列走坏 AND 曾经多头 AND CLOSE>MA60;
MA20结构转弱显示:=MA20结构转弱 AND COUNT(REF(MA20结构转弱,1),5)=0;

{=== 结构转弱 2：转粘合（粘） ===}
近10日多头排列:=COUNT(MA5>MA20 AND MA20>MA60,10)>=5;
今日均线粘合:=均线粘合;
MA20走平下压:=MA20<=REF(MA20,1)*1.002;
转粘合:=近10日多头排列 AND 今日均线粘合 AND MA20走平下压 AND 曾经多头;
转粘合显示:=转粘合 AND COUNT(REF(转粘合,1),5)=0;

{=== 结构转弱 3：趋势破坏（破） ===}
失守MA60:=CLOSE<MA60*0.98;
近5日跌破MA60:=COUNT(CROSS(MA60,CLOSE),5)>0;
MA60走平下压:=MA60<=REF(MA60,1)*1.001;
趋势破坏:=失守MA60 AND 近5日跌破MA60 AND MA60走平下压 AND 曾经多头;
趋势破坏显示:=趋势破坏 AND COUNT(REF(趋势破坏,1),5)=0;

{=== 结构转弱 4：急破 ===}
急跌:=涨跌幅<-3;
急跌放额:=AMT0>均额20*1.5;
急跌收低:=收盘位置<=0.35;
向下跳空:=OPEN<REF(LOW,1);
跌破MA20急:=CLOSE<MA20*0.98;
跌破昨低:=CLOSE<REF(LOW,1);
急破:=高位 AND 急跌 AND 急跌放额 AND 急跌收低 AND (向下跳空 OR 跌破MA20急 OR 跌破昨低);
急破显示:=急破 AND COUNT(REF(急破,1),5)=0;

{=== 结构转弱层内部互斥 ===}
趋势破坏最终:=趋势破坏显示;
急破最终:=急破显示 AND 趋势破坏最终=0;
MA20结构转弱最终:=MA20结构转弱显示 AND 趋势破坏最终=0 AND 急破最终=0;
转粘合最终:=转粘合显示 AND 趋势破坏最终=0 AND 急破最终=0 AND MA20结构转弱最终=0;

{=== 形态预警层：形警 ===}
形态收弱:=收盘位置<=0.45;
近5日前高:=REF(HHV(HIGH,5),1);
近高冲高未站稳:=HIGH>=近5日前高 AND CLOSE<=近5日前高*1.005;
形态短线松动:=CLOSE<MA5 OR MA5<=REF(MA5,1)*1.002 OR 近高冲高未站稳;
射击星警:=高位 AND 形态短线松动 AND 上影>实体*2 AND 上影占比>=0.45 AND 形态收弱 AND 实体/全幅<=0.35;

昨全幅:=REF(HIGH,1)-REF(LOW,1)+0.001;
昨实体:=ABS(REF(CLOSE,1)-REF(OPEN,1));
昨收盘位置:=(REF(CLOSE,1)-REF(LOW,1))/昨全幅;
昨阳有效:=REF(CLOSE,1)>REF(OPEN,1) AND 昨实体/昨全幅>=0.25 AND 昨收盘位置>=0.60;
今阴:=CLOSE<OPEN;
今日反杀:=今阴 AND OPEN>=REF(CLOSE,1)*0.995 AND CLOSE<=REF(OPEN,1)*0.995;
吞没力度:=实体>=昨实体*1.2 AND 实体/全幅>=0.45 AND 收盘位置<=0.35;
吞没警:=高位 AND 形态短线松动 AND 昨阳有效 AND 今日反杀 AND 吞没力度;

昨强阳:=REF(CLOSE,1)>REF(OPEN,1) AND 昨实体/昨全幅>=0.35 AND 昨收盘位置>=0.65;
今日高开冲高:=OPEN>=REF(CLOSE,1)*1.005 OR HIGH>=REF(HIGH,1)*1.01;
昨阳实体中位:=(REF(OPEN,1)+REF(CLOSE,1))/2;
乌云压回:=今阴 AND CLOSE<昨阳实体中位 AND CLOSE>REF(OPEN,1)*0.995 AND 收盘位置<=0.45;
乌云警:=高位 AND 形态短线松动 AND 昨强阳 AND 今日高开冲高 AND 乌云压回;

前日全幅:=REF(HIGH,2)-REF(LOW,2)+0.001;
前日实体:=ABS(REF(CLOSE,2)-REF(OPEN,2));
前日收盘位置:=(REF(CLOSE,2)-REF(LOW,2))/前日全幅;
前日强阳:=REF(CLOSE,2)>REF(OPEN,2) AND 前日实体/前日全幅>=0.40 AND 前日收盘位置>=0.65;
昨日全幅:=昨全幅;
昨日实体:=昨实体;
昨日小星:=昨日实体/昨日全幅<=0.30 AND REF(HIGH,1)>=REF(CLOSE,2)*0.995 AND REF(LOW,1)>=REF(OPEN,2)*0.98;
今日阴确:=今阴 AND 实体/全幅>=0.35 AND 收盘位置<=0.40 AND CLOSE<=(REF(OPEN,2)+REF(CLOSE,2))/2;
黄昏星警:=高位 AND 形态短线松动 AND 前日强阳 AND 昨日小星 AND 今日阴确;

乌云警最终:=乌云警 AND 吞没警=0;
黄昏形最终:=黄昏星警;
吞没形最终:=吞没警 AND 黄昏形最终=0;
乌云形最终:=乌云警最终 AND 黄昏形最终=0 AND 吞没形最终=0;
射击形最终:=射击星警 AND 黄昏形最终=0 AND 吞没形最终=0 AND 乌云形最终=0;
形态层有信号:=黄昏形最终 OR 吞没形最终 OR 乌云形最终 OR 射击形最终;

结构层有信号:=趋势破坏最终 OR 急破最终 OR MA20结构转弱最终 OR 转粘合最终;
