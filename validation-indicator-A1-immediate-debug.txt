日线:=PERIOD=5;
周线:=PERIOD=6;
月线:=PERIOD=7;

有效周期:=日线 OR 周线 OR 月线;
足够K:=BARSCOUNT(C)>IF(周线,80,IF(月线,48,120));
显示许可:=有效周期 AND 足够K;

BBN:=IF(周线,13,IF(月线,12,20));
KCN:=IF(周线,13,IF(月线,12,20));
MAN:=IF(周线,8,IF(月线,6,13));

P:=2;
KCP:=1.5;
深限:=IF(周线,0.25,IF(月线,0.20,0.30));

TN:=IF(周线,26,IF(月线,24,60));
TM:=IF(周线,5,IF(月线,3,10));

WM:=IF(周线,26,IF(月线,24,60));
WRP:=0.85;

SQN:=IF(周线,3,IF(月线,3,5));
VN:=IF(周线,13,IF(月线,12,20));
VP:=1.2;
BVP:=1.8;
BP:=0.45;
YP:=0.45;

BRN:=IF(周线,13,IF(月线,12,20));
FN:=IF(周线,3,IF(月线,2,5));

黄色冷却N:=IF(周线,4,IF(月线,3,8));
黄色趋势段N:=IF(周线,8,IF(月线,5,15));
黄新鲜N:=IF(周线,5,IF(月线,3,10));

败冷却N:=IF(周线,3,IF(月线,2,5));
速冷却N:=IF(周线,4,IF(月线,3,8));
速后N:=IF(周线,5,IF(月线,3,12));
冲冷却N:=IF(周线,4,IF(月线,3,8));

{一、布林带与PCTB}
MID:=MA(C,BBN);
TMP:=STD(C,BBN);
BUP:=MID+P*TMP;
BDN:=MID-P*TMP;

PCTB:=(C-BDN)/MAX(BUP-BDN,0.0001);
图顶:1.30,COLORGRAY,DOTLINE;
图底:-0.30,COLORGRAY,DOTLINE;
PCTB线:MIN(MAX(PCTB,-0.30),1.30),COLORYELLOW,LINETHICK2;

{紧跟输出的固定绘图测试}
早测:=C>0;
STICKLINE(早测,1.05,1.20,3,0),COLORRED;
DRAWTEXT(早测,1.20,'早测'),COLORYELLOW;
